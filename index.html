    <script>

        // 전역 상태 관리 객체

        const state = {

            currentShape: null,     // 현재 선택한 도형 종류 (oval, rectangle, diamond, document)

            currentText: '',        // 도형에 삽입할 텍스트 내용

            isPlacing: false,       // 현재 미리보기 및 배치 활성화 상태 여부

            placedShapes: [],       // 이미 배치 완료된 도형들의 데이터 목록

            lastPlacedShape: null   // 화살표 자동 연결을 위한 직전 배치 도형 데이터

        };

        // DOM 요소 캐싱

        const canvas = document.getElementById('canvas');

        const shapeLayer = document.getElementById('shape-layer');

        const arrowLayer = document.getElementById('arrow-layer');

        const previewContainer = document.getElementById('preview-container');

        const previewShape = document.getElementById('preview-shape');

        const previewArrowPath = document.getElementById('preview-arrow-path');

        const textModal = document.getElementById('text-modal');

        const modalContent = document.getElementById('modal-content');

        const shapeTextInput = document.getElementById('shape-text-input');

        // 도형별 스타일 서식 매핑 테이블

        const shapeStyles = {

            oval: "rounded-full w-32 h-16",

            rectangle: "rounded-md w-32 h-16",

            diamond: "w-24 h-24 rotate-45",

            document: "w-32 h-16 rounded-b-[20%_10%] rounded-t-sm"

        };

        // 1. 도형 선택 이벤트 바인딩

        document.querySelectorAll('#shape-selectors button').forEach(button => {

            button.addEventListener('click', () => {

                state.currentShape = button.getAttribute('data-shape');

                openModal();

            });

        });

        // 2. 모달 제어 로직

        function openModal() {

            textModal.classList.remove('hidden');

            setTimeout(() => {

                modalContent.classList.remove('scale-95', 'opacity-0');

                shapeTextInput.focus();

            }, 50);

        }

        function closeModal() {

            modalContent.classList.add('scale-95', 'opacity-0');

            setTimeout(() => {

                textModal.classList.add('hidden');

                shapeTextInput.value = '';

            }, 200);

        }

        document.getElementById('modal-cancel').addEventListener('click', closeModal);

       

        document.getElementById('modal-confirm').addEventListener('click', () => {

            state.currentText = shapeTextInput.value.trim() || ' ';

            closeModal();

            startPlacingMode();

        });

        shapeTextInput.addEventListener('keydown', (e) => {

            if (e.key === 'Enter') document.getElementById('modal-confirm').click();

        });

        // 3. 배치 모드(반투명 미리보기) 활성화

        function startPlacingMode() {

            state.isPlacing = true;

            previewContainer.classList.remove('hidden');

            // 미리보기 도형 디자인 세팅

            previewShape.className = `absolute border-2 border-slate-600 bg-slate-100 shadow-md flex items-center justify-center shape-text ${shapeStyles[state.currentShape]}`;

           

            // 마름모(조건)의 경우 텍스트 회전 역보정

            if (state.currentShape === 'diamond') {

                previewShape.innerHTML = `<div class="-rotate-45">${state.currentText}</div>`;

            } else {

                previewShape.innerHTML = state.currentText;

            }

        }

        // 4. 작업 화면 내 마우스 이동에 따른 미리보기 추적

        document.addEventListener('mousemove', (e) => {

            if (!state.isPlacing) return;

            const rect = canvas.getBoundingClientRect();

            // 스크롤 및 캔버스 내부 상대 좌표 연산

            const mouseX = e.clientX - rect.left + canvas.scrollLeft;

            const mouseY = e.clientY - rect.top + canvas.scrollTop;

            // 마우스 커서 중심에 맞춘 상대 좌표 오프셋 설정

            const shapeW = state.currentShape === 'diamond' ? 96 : 128;

            const shapeH = state.currentShape === 'diamond' ? 96 : 64;

            const currentX = mouseX - (shapeW / 2);

            const currentY = mouseY - (shapeH / 2);

            // 캔버스 범위를 벗어나면 미리보기를 보이지 않음

            if (mouseX < 0 || mouseX > rect.width || mouseY < 0 || mouseY > rect.height) {

                previewContainer.classList.add('hidden');

                return;

            } else {

                previewContainer.classList.remove('hidden');

            }

            // 도형 렌더링 위치 설정

            previewShape.style.left = `${currentX}px`;

            previewShape.style.top = `${currentY}px`;

            // 직전 도형이 존재한다면 자동 첨가될 반투명 화살표 미리 계산 및 출력

            if (state.lastPlacedShape) {

                const startX = state.lastPlacedShape.centerX;

                const startY = state.lastPlacedShape.centerY;

               

                // 패스 좌표 동적 계산 (직전 중심점 -> 현재 마우스 실시간 좌표)

                previewArrowPath.setAttribute('d', `M ${startX} ${startY} L ${mouseX} ${mouseY}`);

            } else {

                previewArrowPath.setAttribute('d', '');

            }

        });

        // 5. 작업 화면 클릭 시 도형 및 화살표 최종 안착 고정

        canvas.addEventListener('click', (e) => {

            if (!state.isPlacing) return;

            const rect = canvas.getBoundingClientRect();

            const mouseX = e.clientX - rect.left + canvas.scrollLeft;

            const mouseY = e.clientY - rect.top + canvas.scrollTop;

            const shapeW = state.currentShape === 'diamond' ? 96 : 128;

            const shapeH = state.currentShape === 'diamond' ? 96 : 64;

            const finalX = mouseX - (shapeW / 2);

            const finalY = mouseY - (shapeH / 2);

            // 새 도형 객체 데이터 구성

            const newShape = {

                id: 'shape_' + Date.now(),

                type: state.currentShape,

                text: state.currentText,

                x: finalX,

                y: finalY,

                width: shapeW,

                height: shapeH,

                centerX: mouseX,

                centerY: mouseY

            };

            // 만약 이전 배치 노드가 존재한다면 고정식 SVG 화살표 레이어 추가 선행

            if (state.lastPlacedShape) {

                const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');

                path.setAttribute('d', `M ${state.lastPlacedShape.centerX} ${state.lastPlacedShape.centerY} L ${newShape.centerX} ${newShape.centerY}`);

                path.setAttribute('fill', 'none');

                path.setAttribute('stroke', '#475569');

                path.setAttribute('stroke-width', '2.5');

                path.setAttribute('marker-end', 'url(#head)'); // 실선 마커 적용을 위해 소문자 head 세팅 하단 진행

                arrowLayer.appendChild(path);

            }

            // 실제 DOM 엘리먼트로 캔버스 내 정식 추가 생성

            const shapeDiv = document.createElement('div');

            shapeDiv.className = `absolute border-2 border-slate-800 bg-white shadow-sm flex items-center justify-center shape-text ${shapeStyles[newShape.type]}`;

            shapeDiv.style.left = `${newShape.x}px`;

            shapeDiv.style.top = `${newShape.y}px`;

            if (newShape.type === 'diamond') {

                shapeDiv.innerHTML = `<div class="-rotate-45">${newShape.text}</div>`;

            } else {

                shapeDiv.innerHTML = newShape.text;

            }

            shapeLayer.appendChild(shapeDiv);

            // 상태 저장 업데이트

            state.placedShapes.push(newShape);

            state.lastPlacedShape = newShape;

            // 배치 제어 플래그 해제 및 프리뷰 컨테이너 은닉화 처리

            state.isPlacing = false;

            previewContainer.classList.add('hidden');

            previewArrowPath.setAttribute('d', '');

        });

        // 6. 새로 만들기 초기화 리셋 스크립트

        document.getElementById('btn-new').addEventListener('click', () => {

            if (confirm('작업 중인 모든 내용이 삭제됩니다. 새로 만드시겠습니까?')) {

                shapeLayer.innerHTML = '';

                // SVG 초기화 진행하되 marker 요소 정의 구조는 보존

                arrowLayer.innerHTML = `

                    <defs>

                        <marker id="arrowhead" markerWidth="10" markerHeight="7" refX="8" refY="3.5" orient="auto">

                            <polygon points="0 0, 10 3.5, 0 7" fill="#475569" />

                        </marker>

                        <marker id="head" markerWidth="10" markerHeight="7" refX="18" refY="3.5" orient="auto">

                            <polygon points="0 0, 10 3.5, 0 7" fill="#475569" />

                        </marker>

                    </defs>

                `;

                canvas.style.backgroundImage = 'none';

                state.placedShapes = [];

                state.lastPlacedShape = null;

                state.isPlacing = false;

            }

        });

        // SVG 마커 정의 보강 (실선용 패딩 여유값 확보)

        window.addEventListener('DOMContentLoaded', () => {

            const defs = arrowLayer.querySelector('defs');

            const headMarker = document.createElementNS('http://www.w3.org/2000/svg', 'marker');

            headMarker.setAttribute('id', 'head');

            headMarker.setAttribute('markerWidth', '10');

            headMarker.setAttribute('markerHeight', '7');

            // 도형 안쪽으로 너무 파고들지 않도록 오프셋 값 조정

            headMarker.setAttribute('refX', '38');

            headMarker.setAttribute('refY', '3.5');

            headMarker.setAttribute('orient', 'auto');

            headMarker.innerHTML = '<polygon points="0 0, 10 3.5, 0 7" fill="#475569" />';

            defs.appendChild(headMarker);

        });

        // 7. 로컬 백그라운드 이미지 스캔 불러오기

        document.getElementById('btn-load').addEventListener('change', (e) => {

            const file = e.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = function(event) {

                    canvas.style.backgroundImage = `url('${event.target.result}')`;

                };

                reader.readAsDataURL(file);

            }

        });
